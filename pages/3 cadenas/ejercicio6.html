Contar caracteres especificos

codigo : javascript

let cadena = "Hola Mundo";
let caracter = "o";
let contador = 0; //1 Declarar las variables cadena, caracter y contador.
for (let i = 0; i < cadena.length; i++) { //2 Usar un bucle for para recorrer la cadena.
if (cadena[i] === caracter) { //3 Si el carácter actual es igual a caracter, incrementar el contador.
contador++;
}
}
