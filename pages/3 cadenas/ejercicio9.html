Eliminar espacios

codigo : javascript
//1.Declarar la variable cadena.
let cadena = "Hola Mundo desde Javascript";
//2.Declarar la variable cadenaSinEspacios.
let cadenaSinEspacios = "";
//3.Usar un bucle for para recorrer cada carácter de la cadena.
for (let i = 0; i < cadena.length; i++) {
//4.Si el carácter actual no es un espacio, añadirlo a cadenaSinEspacios.
if (cadena[i] !== " ") {
cadenaSinEspacios += cadena[i];
}
}
console.log(cadenaSinEspacios);
