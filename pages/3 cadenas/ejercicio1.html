Imprimir cada carácter de una cadena

codigo : javascript
//1.Declarar una variable para la cadena.
let cadena = "Hola Mundo";
//2.Itera desde 0 hasta cadena.length - 1.
for(let i = 0; i < cadena.length; i++) {
//3.Imprime el carácter actual.
console.log(cadena[i]);
}
