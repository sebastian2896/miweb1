Mayuscula A Minuscula y Viceversa

codigo : javascript

let cadena = "Hola Mundo";
let nuevaCadena = ""; //1 Declarar las variables cadena y nuevaCadena.
for (let i = 0; i < cadena.length; i++) { //2 Usar un bucle for para recorrer cada carácter de la cadena.
let char = cadena[i]; //3 Usar char.charCodeAt(0) para obtener el código ASCII del carácter y ajustar el valor para convertir entre mayúsculas y minúsculas.
if (char >= 'a' && char <= 'z') {
nuevaCadena += String.fromCharCode(char.charCodeAt(0) - 32);
} else if (char >= 'A' && char <= 'Z') {
nuevaCadena += String.fromCharCode(char.charCodeAt(0) + 32);
} else {
nuevaCadena += char;
}
}
