Contador de palabras

codigo : javascript

let cadena = "Hola Mundo desde Javascript";
let contadorPalabras = 1; // Asumimos que hay al menos una palabra
for (let i = 0; i < cadena.length; i++) { //2 Usar un bucle for para recorrer la cadena.
if (cadena[i] === " ") { //3 Si el carácter actual es un espacio, incrementar contadorPalabras.
contadorPalabras++;
}
}
console.log(contadorPalabras);
