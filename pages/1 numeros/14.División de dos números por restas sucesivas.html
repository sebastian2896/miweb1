Division por resta sucesivas

codigo : javascript

let dividendo = 20;
let divisor = 4;
let cociente = 0;
while (dividendo >= divisor) { //1 itera mientras el dividendo sea mayor o igual al divisor
dividendo -= divisor; //2 resta el divisor del dividendo
cociente++; //3 incrementa el cociente en 1 en cada iteración
}
console.log("El cociente de la división es " + cociente);
