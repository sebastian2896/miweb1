Determinar si un año es bisiesto

codigo : javascript

let año = document.getElementById('año').value
año = parseInt(año)
let resultadoAño = document.getElementById('resp')
resultadoAño.innerHTML = ''
let explicacion
if ((año % 4 === 0 && año % 100 !== 0) || año % 400 === 0) {
explicacion = 'Si es bisiesto porque el numero del año al cumplir la condicion de ser divisible para 4 y 100 pero no a 100';
}
else {
explicacion = 'No es bisiesto porque el numero del año al no cumplir la condicion de ser divisible para 4 y 100 pero no a 100';
}
resultadoAño.innerHTML = explicacion
}
