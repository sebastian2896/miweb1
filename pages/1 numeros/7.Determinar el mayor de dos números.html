Determinar el mayor de dos números

codigo : javascript
let numero1 = 5;
let numero2 = 7;
//1.verifica si el primer número es mayor
if (numero1 > numero2) {
console.log(numero1 + " es mayor que " + numero2);
//2.verifica si el segundo numero es mayor
} else if (numero1 < numero2) {
console.log(numero2 + " es mayor que " + numero1);
} else {
//3. Si ninguna de las condiciones anteriores es verdadera, ambos números son iguales.
console.log("Los números son iguales");
}
