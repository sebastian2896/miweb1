Invertir un Numero

codigo : javascript

let numero = 12345;
let invertido = 0;
//1 itera mientras numero no sea 0
while (numero !== 0) {
//2 obtiene el último dígito del número
let digito = numero % 10;
//3 agrega el dígito al número invertido
invertido = invertido * 10 + digito;
//4 elimina el último dígito del número
numero = Math.floor(numero / 10);
}
console.log("El número invertido es " + invertido);
