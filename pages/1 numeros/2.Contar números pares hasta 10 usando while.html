Contador de Par

codigo : javascript
let i = 0; //declaramos nuestro inicio

let contador = 0; //declaramos nuestro contador

while (i <= 10){ //si nuestro numero inicial es menor o igual a 10 empezara el bucle hasta 10

if (i % 2 === 0) { //Verificara los numeros pares

contador++; //Aqui van los pares

}

i++; //Esto nos ayudara a que nuestro numero aumente mas 1

}

console.log("Hay", contador, "números pares hasta 10"); //los presentamos
