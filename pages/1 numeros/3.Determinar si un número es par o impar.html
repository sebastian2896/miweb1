Determinar si un número es par o impar

codigo : javascript
function par() {
let numero = 4;
numero= document.getElementById('num').value;
numero = parseInt(numero);
let r = document.getElementById('resp');
//1.verifica si el número es divisible por 2.
if (numero % 2 === 0) {

//2.Si la condición es verdadera, el número es par.
console.log(numero + " es par");
r.innerHTML= numero + " es par";
//3.Si no, el número es impar.
} else {
console.log(numero + " es impar");
r.innerHTML= numero + " es impar";
}
}
