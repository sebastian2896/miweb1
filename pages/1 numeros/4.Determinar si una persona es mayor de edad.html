Determinar si es mayor de edad

codigo : javascript

let edad = document.getElementById('edad').value
edad = parseInt(edad)
let explicacion = document.getElementById('explicacion')
explicacion.innerHTML = ''
if(edad >=18){
explicacion += `La edad de ${edad} al cumplir la condicion es mayor de edad`
}
else{
explicacion += `La edad de ${edad} al cumplir no la condicion no es mayor de edad`
}
explicacion.innerHTML = explicacion
}
