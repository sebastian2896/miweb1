Múltiplos de 5 en un arreglo

codigo : javascript
//1. Declarar un arreglo de números
let numeros = [10,15,23,25,30,45,50];
let multiplos = [];
//2.itera sobre cada número en el arreglo.
for(let i = 0; i < numeros.length; i++) {
//3.verifica si el número es múltiplo de 5
if(numeros[i] % 5 === 0) {
console.log(numeros[i]+ "es multiplo de 5");
}
}
