Quitar a Ana del arreglo

codigo : javascript

let nombres = ["Juan", "Ana", "Carlos", "Ana", "Luis"];
for (let i = 0; i < nombres.length; i++) { //1 itera a través del arreglo
if (nombres[i] !== "Ana") { //2 verifica si el nombre no es "Ana".
console.log(nombres[i]);
}
}
